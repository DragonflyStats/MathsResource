	\begin{multicols}{2}
		\begin{itemize}
			\item[(a)] A does not hit the target, 
			\item[(b)] both hit the target, 
			\item[(c)] only one of them hits the target,
			\item[(d)] at least one hits the target
			\item[(e)] neither hits the target.\\
		\end{itemize}
	\end{multicols}
	
\end{enumerate}
