\frametitle{Integration by Parts}
\LARGE
\begin{itemize}
\item Starting with two expressions $u$ and $dv$
\end{itemize}

\[\int u dv = uv - \int vdu \]
\begin{itemize}
\item $v$ is evaluated by integrating $dv$
\item $du$ is evaluated by differentiating $u$.
\end{itemize}
