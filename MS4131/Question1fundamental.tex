	\begin{matrix}
	a & 0 & 0 \\
	b & c & 0 \\
	d & e & f \\
	\end{matrix}
	\]

			\item[(iii)] A can be expressed as the sum of a symmetric matrix, $\frac{1}{2}(A+A^T)$ and a skew

		\end{itemize}
