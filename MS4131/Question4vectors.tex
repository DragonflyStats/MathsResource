		2x_1 + 4x_2 + x_3 &= -3\\

		\end{align*}

			\item Write down the coefficient matrix and the augmented matrix of this system. \marks{4}

		\end{enumerate}

		x_1 + x_3 &= 0\\
		2x_1 + 4x_2 + x_3 &= 0\\
		x_2 + 3x_3 &= 0.
		\end{align*}
		What can you say about its solution set?\marks{4}
	\end{enumerate}
